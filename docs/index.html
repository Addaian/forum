              How do I run the server?
            
            
# 1. install (one time)
uv pip install -e ".[server]"

# 2. set secrets
export ANTHROPIC_API_KEY=sk-...
export FORUM_API_TOKEN=$(openssl rand -hex 16)   # share this with your friends

# 3. boot
uv run uvicorn server:app --port 8001

# 4. expose to the internet (pick one)
cloudflared tunnel --url http://localhost:8001   # zero account, random URL
ngrok http 8001                                  # stable URL with free account

# 5. paste the tunnel URL + token into this dialog
